highlight[string code] → string (string)

Returns a string with ANSII escape sequences surrounding each token in code. 
highlight := [code] -> {
    tokens .= Tokenize[code]
    inner .= Join! {
        Format["\x1b[%sm%s\x1b[0m", highlight_styles[_.type], _.raw]
    } => tokens
}
