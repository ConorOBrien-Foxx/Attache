HeatMap[mat, legend, min, max] := (
    min ..= Min[mat];;
    max ..= Max[mat];;
    legend ..= FULL_LEGEND;;
    diffs .= If[max = min,
        mat * 0,
        (mat - min) / (max - min)
    ];;
    MultiJoin <| legend[diffs * (#legend - 1) | Floor]
)

PadMatrix[ragged, padwith, padf] := (
    padwith ..= 0;;
    padf ..= PadRight;;
    maxlen .= Max[Size => ragged];;
    padf<~_, maxlen, padwith~> => ragged
)
